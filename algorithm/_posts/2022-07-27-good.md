    end = 0

            break
        while summary < m and end < N-1:
            if start == end:
                summary = check[start]
            else:
                summary = check[start]+check[end]
            end +=1
        if summary == m:
            good_count += 1
            break

print(good_count)

import sys
input = sys.stdin.readline

N = int(input())
arr = list(map(int,input().split()))

arr.sort()
good_count = 0

for i in range(N):
    check = arr[:i] + arr[i+1:]
    start,end = 0,N-2
    sum = 0
    target = arr[i]

    while start < end:
        sum = check[start]+check[end]

        if sum == target:
            good_count += 1
            break
        
        if sum > target:
            end -= 1
        
        if sum < target:
            start += 1

print(good_count)
